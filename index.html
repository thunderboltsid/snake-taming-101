Python conditionals

                    
>>> def fibonacci(n):
        if (n < 0):
            raise Exception("Negative fibonacci numbers don't exist")
        elif (n == 0 or n == 1):
            return 1
        else:
            return fibonacci(n-1) + fibonacci(n-2)

Inheritance

                    
class Parent(object):
    def override(self):
        print("PARENT override()")

    def implicit(self):
        print("PARENT implicit()")

    def altered(self):
        print("PARENT altered()")

class Child(Parent):
    def override(self):
        print("CHILD override()")

    def altered(self):
        print("CHILD, BEFORE PARENT altered()")
        super(Child, self).altered()
        print("CHILD, AFTER PARENT altered()")

dad = Parent()
son = Child()

son.implicit()
son.override()
son.altered()

Multiple Inheritance

                    
class First(object):
    def __init__(self):
        print ("first")

class Second(First):
    def __init__(self):
        print ("second")

class Third(First):
    def __init__(self):
        print ("third")

class Fourth(Second, Third):
    def __init__(self):
        super(Fourth, self).__init__()
        print ("that's it")

Fourth()

Multiple Inheritance (BAD!!!)

                    
class First(object):
    def __init__(self):
        print ("first")

class Second(First):
    def __init__(self):
        print ("second")

class Third(First, Second):
    def __init__(self):
        print ("third")

Third() # Oops!

List Comprehension?

                    
numbers = [1, 2, 3, 4, 5]

doubled_odds = []
for n in numbers:

        doubled_odds.append(n * 2)

List Comprehension!

                    
numbers = [1, 2, 3, 4, 5]

doubled_odds = [n * 2 for n in numbers if n % 2 == 1]

Decorators

                
def my_decorator(a_function_to_decorate):
    def the_wrapper_around_the_original_function():
        print("Before the function runs")
        a_function_to_decorate()
        print("After the function runs")
    return the_wrapper_around_the_original_function


@my_decorator
def stand_alone_function():
    print("Leave me alone")

stand_alone_function() #is the same as
my_decorator(stand_alone_function())
